217. Contains Duplicate - 1 minute
class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        hashSet = set()

        for n in nums:
            if n in hashSet:
                return True
            else:
                hashSet.add(n)

        return False


1. Two Sum  - 3 mintues

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        hashmap = {}

        for i, num in enumerate(nums):
            k = target - num
            if k in hashmap:
                return [hashmap[k], i]
            else:
                hashmap[num] = i
        
        return []

        #given t

        s = s.lower()
        t = t.lower()
        count = [0]*26

        #add to count in corresponding index as we see characters

            count[idx] += 1
        print(count)
        
        #subtract from count in corresponding index as we see characters
        for ch in t:
            idx = ord(ch) - ord("a")
            count[idx] -= 1
        print(count)

        #if all counts are same, at the end, all entries should be 0, else return False

            if count[i] != 0:

14. Longest Common Prefix  - 3 minutes or so (sorting hint from last revision with GPT)
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        #we will sort the strs for ease
        # print(strs)
        strs = sorted(strs)      #(O(loe(n)))
        n = len(strs)
        # print(strs)
        res = ""
        for i, ch in enumerate(strs[0]):
            if ch == strs[n-1][i]:
                res += ch
            else:
                return res
        
        return res

49. Group Anagrams - 10 minutes 
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        #use the sorted of the strs as keys, and keep appending them in the values, which is a list. 
        #at the end, return the values from the hashmap
        hashmap = {}
        for word in strs:
            key = tuple(sorted(word))     #list can't be a key to hashmap, so tuple()
            # print(key, word)
            if key in hashmap:
                hashmap[key].append(word)
            else:
                hashmap[key] = [word]
        
        return list(hashmap.values())

27. Remove Element - 2 minutes
class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        #given nums
        #given val --> remove all val in nums in place. 
        #return # elements != val, i.e. return len(nums) at last

        n = len(nums)
        i = 0
        while i < n:
            if nums[i] == val:
                del nums[i]
            else:
                i += 1
            n = len(nums)

        return len(nums)


169. Majority Element
class Solution: 
    def majorityElement(self, nums: List[int]) -> int:
        #given nums of size n
        n = len(nums)
        count = {}
        for num in nums:
            count[num] = count.get(num, 0) + 1
            if count[num] > n // 2:
                return num


705. Design HashSet
class MyHashSet:

    def __init__(self):
        self.hashset = set()

    def add(self, key: int) -> None:
        self.hashset.add(key)

    def remove(self, key: int) -> None:
        if key in self.hashset:
            self.hashset.remove(key)

    def contains(self, key: int) -> bool:
        return key in self.hashset

706. Design HashMap
class MyHashMap:

    def __init__(self):
        self.hashmap = dict()

    def put(self, key: int, value: int) -> None:
        self.hashmap[key] = value

    def get(self, key: int) -> int:
        if key in self.hashmap:
            return self.hashmap[key]
        else:
            return -1

    def remove(self, key: int) -> None:
        if key in self.hashmap:
            del self.hashmap[key]
