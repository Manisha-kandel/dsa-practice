217. Contains Duplicate - 1 minute
class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        hashSet = set()

        for n in nums:
            if n in hashSet:
                return True
            else:
                hashSet.add(n)

        return False


1. Two Sum  - 3 mintues

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        hashmap = {}

        for i, num in enumerate(nums):
            k = target - num
            if k in hashmap:
                return [hashmap[k], i]
            else:
                hashmap[num] = i
        
        return []

49. Group Anagrams - 10 minutes 
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        #use the sorted of the strs as keys, and keep appending them in the values, which is a list. 
        #at the end, return the values from the hashmap
        hashmap = {}
        for word in strs:
            key = tuple(sorted(word))     #list can't be a key to hashmap, so tuple()
            # print(key, word)
            if key in hashmap:
                hashmap[key].append(word)
            else:
                hashmap[key] = [word]
        
        return list(hashmap.values())

27. Remove Element - 2 minutes
class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        #given nums
        #given val --> remove all val in nums in place. 
        #return # elements != val, i.e. return len(nums) at last

        n = len(nums)
        i = 0
        while i < n:
            if nums[i] == val:
                del nums[i]
            else:
                i += 1
            n = len(nums)

        return len(nums)

        #populate the counter
        for n in nums:

        heapq.heapify(top_k)

75. Sort Colors  - 10 minutes
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        #they are colored red, white or bluw. 
        #need to sort in place s.t. same colors are adjacent
        '''
        Red = 0
        white = 1
        blue = 2
        '''

        #let's use two pointers l and r to swap in place  for 0 and 2, when those two are done, 1's will be automatically sorted

        l, r = 0, len(nums)-1

        for i in range(len(nums)):
            if nums[i] == 0:
                nums[i], nums[l] = nums[l], nums[i]
                l += 1
        
        for i in range(len(nums)):
            if nums[i] == 1:
                nums[i], nums[l] = nums[l], nums[i]      #use the continued l
                l += 1

        return nums


304. Range Sum Query 2D - Immutable   - 25 minutes (15 + 10)
class NumMatrix:

    def __init__(self, matrix: List[List[int]]):
        self.matrix = matrix
        rows = len(self.matrix)
        cols = len(self.matrix[0])

        #initiate prefix matrix (with extra row and col in from to keep zeros to avoid index error)
        self.prefix = [[0]*(cols + 1) for _ in range(rows + 1)]

        for i in range(1, rows + 1):
            for j in range(1, cols + 1):
                self.prefix[i][j] = (self.matrix[i-1][j-1] 
                                    + self.prefix[i-1][j]
                                    + self.prefix[i][j-1]
                                    - self.prefix[i-1][j-1])
                                    


    def sumRegion(self, row1: int, col1: int, row2: int, col2: int) -> int:
        #since the original rows and cols are shifted by 1 position due to 0 col and row at first
        r1 = row1 + 1
        r2 = row2 + 1
        c1 = col1 + 1
        c2 = col2 + 1

        return (self.prefix[r2][c2] 
                - self.prefix[r2][c1 - 1]
                - self.prefix[r1-1][c2]
                + self.prefix[r1-1][c1-1])

238. Product of Array Except Self  - 8 minutes
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        prefix = [1]* n
        suffix = [1]* n
        
        #populate prefix
        pProduct = 1
        i = 0
        while i < n:
            pProduct *= nums[i]
            prefix[i] = pProduct
            i += 1

        
        #populate suffix
        sProduct = 1
        i = n - 1
        while i >= 0:
            sProduct *= nums[i]
            suffix[i] = sProduct
            i -= 1
        
        #make res and populate
        res = [1]*n
        res[0] = suffix[1]
        res[n-1] = prefix[n-2]

        for i in range(1, n-1):
            res[i] = prefix[i-1] * suffix[i+1]

        return res



36. Valid Sudoku- 10 minutes
class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        #check if it's row, col and box already has the number or not
        rows = [[False]*9 for i in range(9)]
        cols = [[False]*9 for i in range(9)]
        boxes = [[[False]*9 for i in range(3)] for j in range(3)]

        # print(len(rows), len(rows[0]))
        # print(len(cols), len(cols[0]))
        # print(len(boxes), len(boxes[0]), len(boxes[0][0]))

        for i in  range(9):
            for j in range(9):
                if board[i][j] != ".":
                    num = int(board[i][j])
                    if rows[i-1][num-1] == True:
                        return False
                    if cols[j-1][num-1] == True:
                        return False
                    if boxes[(i)//3][(j)//3][num-1] == True:
                        return False
                    
                    rows[i-1][num-1] = True
                    cols[j-1][num-1] = True
                    boxes[(i)//3][(j)//3][num-1] = True
        
        return True


128. Longest Consecutive Sequence - 8 mins
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        # we first make a set of nums
        #then keep checking for all the number n in that set, if n-1 is not present in the set we start taking count of that, and update the res accordingly if the length of the consecutive is longer than the res.

        hashset = set(nums)
        res = 0
        # print(hashset)

        # print( nums[0] in hashset) 
        for n in hashset:      #
            if n-1 not in hashset:
                length = 1

                while n + length in hashset:
                    length += 1
                
                res = max(res, length)
        
        return res


122. Best Time to Buy and Sell Stock II - 2 minutes (prettty simple)
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        #think of it as a graph, we need to find the rising part and keep adding that rise, that's all
        profit = 0
        for i in range(1, len(prices)):
            diff = prices[i] - prices[i-1]
            if diff > 0:
                profit += diff

        return profit


229. Majority Element II - 5 mins: there must be a better solution.
class Solution:
    def majorityElement(self, nums: List[int]) -> List[int]:
        #we make hash table and keep adding elements to res, if the count crosses n//3 times, after crossing, there is no need to keep updating them for the count as well

        count = {}
        res = []
        length = len(nums)
        for n in nums:
            if n not in res:
                count[n] = count.get(n, 0) + 1

                if count[n] > int(length/3):
                    res.append(n)
        
        return res


560. Subarray Sum Equals K - 15 minutes (0 + 13 + 2)
class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        #concept of prefixsum and hashmap will be used here. we keep track of prefixsum count in a hashmap
        #then using that, we find out if there is a subarray sum equal to the target in the array and correspondingly, also update the res by the #times the subarray sum is possible. 

        if not nums: return 0 

        prefix_count = {0:1}   #important base case
        
        prefix = [0]*len(nums)
        pSum = 0
        res = 0
        for i, num in enumerate(nums):
            pSum += num
            prefix[i] = pSum
            

            target = pSum - k          
            if target in prefix_count:
                res += prefix_count[target]

            #update at last
            prefix_count[pSum] = prefix_count.get(pSum, 0) + 1
            
        return res
